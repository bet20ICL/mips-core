module data_memory(
    input logic clk,
    input[31:0] adress,
    input memwrite,
    input[31:0] data_in,
    output[31:0] data_out

);

    reg [31:0] ram [65535:0];
    integer i;
    
    initial begin
        for(i=0;i<65535;i=i+1)
            ram[i] <= 32'd0;
    end
    always @(posedge clk) begin
        if (memwrite)
            ram[adress] <= data_in;
    end
    assign data_out = ram[adress];
endmodule
